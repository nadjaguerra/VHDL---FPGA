COMANDO IF

PORTA OR

architecture orif of estrutura is

signal x: std_logic_vector(1 downto 0);

begin

x <= a & b;

process (a, b)

begin

if (x<="00") then

c<='0';

else

c<='1';

end if;

end process;

end orif;

COMANDO CASE

PORTA OR

architecture caseor of estrutura is

signal x: std_logic_vector (1 downto 0);

begin

x<= a & b;

process (a, b)

begin

 case x is

        when "00" => c <='0';

        when "01" => c <='1';

        when "10" => c <='1';

        when "11" => c <='1';

end case;

end process;

end caseor;
